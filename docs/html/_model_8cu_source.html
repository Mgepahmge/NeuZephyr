87Node* nz::Model::Linear(Node* input, size_t outSize) {
88 auto inputSize = input->output->shape()[1] * input->output->shape()[2] * input->output->shape()[3];
89 Node* shapedInput;
90 if (input->output->shape()[2] != inputSize) {
91 shapedInput = Reshape(input, {input->output->shape()[0], 1, inputSize, 1});
92 }
93 else {
94 shapedInput = input;
95 }
96 auto mulParam = new io::InputNode({1, 1, outSize, inputSize}, true);
97 mulParam->output->randomize();
98 hiddenNodes.push_back(mulParam);
99 computeGraph.addNode(mulParam);
100 auto mulResult = Mul(mulParam, shapedInput);
101 auto biasResult = Bias(mulResult);
102 return biasResult;
103}

204Node* nz::Model::TargetExpand(Node* input, const Tensor::shape_type& shape) {
205 if (!computeGraph.inGraph(input)) {
206 computeGraph.addNode(input);
207 }
208 if (input->output->shape() == shape) {
209 return input;
210 }
211 if (input->output->shape()[0] != 1 ||
212 input->output->shape()[1] != shape[1] ||
213 input->output->shape()[2] != shape[2] ||
214 input->output->shape()[3] != shape[3]) {
215 throw std::runtime_error("The input data cannot be expanded.");
216 }
217 auto* expandNode = new calc::ExpandNode(input, shape.N());
218 hiddenNodes.push_back(expandNode);
219 computeGraph.addNode(expandNode);
220 return expandNode;
221}

244Node* nz::Model::Conv2d(Node* input, Tensor::size_type outChannels, Tensor::size_type kernelHeight,
245 Tensor::size_type kernelWidth, Tensor::size_type stride, Tensor::size_type padding, bool bias) {
246 if (!computeGraph.inGraph(input)) {
247 computeGraph.addNode(input);
248 }
249 auto* convKernel = new io::InputNode({
250 input->output->shape().N(), 1,
251 input->output->shape().C() * kernelHeight * kernelWidth, outChannels
252 }, true);
253 convKernel->output->randomize();
254 hiddenNodes.push_back(convKernel);
255 computeGraph.addNode(convKernel);
256 auto inputCol = Img2Col(input, kernelHeight, kernelWidth, stride, padding);
257 auto resultCol = Mul(inputCol, convKernel);
258 if (bias) {
259 resultCol = Bias(resultCol);
260 }
261 return Col2Img(resultCol, (input->output->shape().H() + 2 * padding - kernelHeight) / stride + 1,
262 (input->output->shape().W() + 2 * padding - kernelWidth) / stride + 1);
263}
