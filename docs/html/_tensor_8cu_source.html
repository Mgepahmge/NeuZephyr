348 if (shape[0] * shape[1] != _size) {
349 WARN("Reshaping to a different size will cause data loss");
350 }
351 auto* temp = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
352 CHECK(cudaMemcpy(temp, _data, _size * sizeof(value_type), cudaMemcpyDeviceToHost));
353 CHECK(cudaFree(_data));
354
355 value_type* temp_grad = nullptr;
356 if (_requires_grad) {
357 temp_grad = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
358 CHECK(cudaMemcpy(temp_grad, _grad, _size * sizeof(value_type), cudaMemcpyDeviceToHost));
359 CHECK(cudaFree(_grad));
360 }
361
362 const size_type size = _size;
363 _size = shape[0] * shape[1];
364 _shape = shape;
365
366 CHECK(cudaMalloc(&_data, _size * sizeof(value_type)));
367 CHECK(cudaMemset(_data, 0, _size * sizeof(value_type)));

379 void Tensor::transpose() {
380 const dim3 block(TILE_SIZE, TILE_SIZE);
381 const dim3 grid((_shape[0] + block.x - 1) / block.x, (_shape[1] + block.y - 1) / block.y);
382 value_type* temp;
383 CHECK(cudaMalloc(&temp, _size * sizeof(value_type)));
384 krnl::Transpose(grid, block, _data, temp, _shape[0], _shape[1]);
385 CHECK(cudaDeviceSynchronize());
386 CHECK(cudaFree(_data));
387 _data = temp;
388 if (_requires_grad) {
389 value_type* tempGrad;
390 CHECK(cudaMalloc(&tempGrad, _size * sizeof(value_type)));
391 krnl::Transpose(grid, block, _grad, tempGrad, _shape[0], _shape[1]);
392 CHECK(cudaDeviceSynchronize());
393 CHECK(cudaFree(_grad));
394 _grad = tempGrad;
395 }
