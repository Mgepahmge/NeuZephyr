709 Tensor Softmax(const Tensor& tensor) {
710 dim3 block(256);
711 dim3 grid((tensor._size + block.x - 1) / block.x);
712 float* result_d;
713 float* result_h;
714 float sum = 0;
715 cudaMalloc(&result_d, grid.x * sizeof(Tensor::value_type));
716 result_h = (float*)malloc(grid.x * sizeof(Tensor::value_type));
717 krnl::SummationExp(grid, block, block.x * sizeof(float), result_d, tensor._data, tensor._size);
718 cudaMemcpy(result_h, result_d, grid.x * sizeof(Tensor::value_type), cudaMemcpyDeviceToHost);
719 for (int i = 0; i < grid.x; i++) {
720 sum += result_h[i];
721 }
722 free(result_h);
723 cudaFree(result_d);
724 krnl::Softmax(grid, block, tensor._data, tensor._data, sum, tensor._size);
725 return tensor;

972 if (shape[0] * shape[1] != _size) {
973 WARN("Reshaping to a different size will cause data loss");
974 }
975 auto* temp = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
976 cudaMemcpy(temp, _data, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
977 cudaFree(_data);
978 value_type* temp_grad = nullptr;
979 if (_requires_grad) {
980 temp_grad = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
981 cudaMemcpy(temp_grad, _grad, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
982 cudaFree(_grad);
983 }
984 const size_type size = _size;
985 _size = shape[0] * shape[1];
986 _shape = shape;
987 cudaMalloc((value_type**)&_data, _size * sizeof(value_type));
988 cudaMemset(_data, 0, _size * sizeof(value_type));
989 cudaMemcpy(_data, temp, size * sizeof(value_type), cudaMemcpyHostToDevice);
990 free(temp);
991 if (_requires_grad) {
992 cudaMalloc((value_type**)&_grad, _size * sizeof(value_type));
993 cudaMemset(_grad, 0, _size * sizeof(value_type));
