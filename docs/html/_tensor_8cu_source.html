711 Tensor Softmax(const Tensor& tensor) {
712 dim3 block(256);
713 dim3 grid((tensor._size + block.x - 1) / block.x);
714 float* result_d;
715 float* result_h;
716 float sum = 0;
717 cudaMalloc(&result_d, grid.x * sizeof(Tensor::value_type));
718 result_h = (float*)malloc(grid.x * sizeof(Tensor::value_type));
719 krnl::SummationExp(grid, block, block.x * sizeof(float), result_d, tensor._data, tensor._size);
720 cudaMemcpy(result_h, result_d, grid.x * sizeof(Tensor::value_type), cudaMemcpyDeviceToHost);
721 for (int i = 0; i < grid.x; i++) {
722 sum += result_h[i];
723 }
724 free(result_h);
725 cudaFree(result_d);
726 krnl::Softmax(grid, block, tensor._data, tensor._data, sum, tensor._size);
727 return tensor;
728 }

972 void Tensor::reshape(const shape_type& shape) {
973 if (shape[0] * shape[1] != _size) {
974 WARN("Reshaping to a different size will cause data loss");
975 }
976 auto* temp = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
977 cudaMemcpy(temp, _data, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
978 cudaFree(_data);
979 value_type* temp_grad = nullptr;
980 if (_requires_grad) {
981 temp_grad = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
982 cudaMemcpy(temp_grad, _grad, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
983 cudaFree(_grad);
984 }
985 const size_type size = _size;
986 _size = shape[0] * shape[1];
987 _shape = shape;
988 cudaMalloc((value_type**)&_data, _size * sizeof(value_type));
989 cudaMemset(_data, 0, _size * sizeof(value_type));
990 cudaMemcpy(_data, temp, size * sizeof(value_type), cudaMemcpyHostToDevice);
991 free(temp);
992 if (_requires_grad) {
993 cudaMalloc((value_type**)&_grad, _size * sizeof(value_type));
994 cudaMemset(_grad, 0, _size * sizeof(value_type));
995 cudaMemcpy(_grad, temp_grad, size * sizeof(value_type), cudaMemcpyHostToDevice);
