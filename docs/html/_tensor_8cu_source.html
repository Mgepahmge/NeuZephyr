411 Tensor Softmax(const Tensor& tensor) {
412 const dim3 block(256);
413 const dim3 grid((tensor._size + block.x - 1) / block.x);
414 float* result_d;
415 float* result_h;
416 float sum = 0;
417 cudaMalloc(&result_d, grid.x * sizeof(Tensor::value_type));
418 result_h = static_cast<float*>(malloc(grid.x * sizeof(Tensor::value_type)));
419 krnl::SummationExp(grid, block, block.x / WARP_SIZE * sizeof(float), result_d, tensor._data, tensor._size);
420 CHECK(cudaMemcpy(result_h, result_d, grid.x * sizeof(Tensor::value_type), cudaMemcpyDeviceToHost));
421 for (int i = 0; i < grid.x; i++) {
422 sum += result_h[i];
423 }
424 free(result_h);
425 cudaFree(result_d);
426 krnl::Softmax(grid, block, tensor._data, tensor._data, sum, tensor._size);
427 CHECK(cudaDeviceSynchronize());
428 return tensor;

674 void Tensor::reshape(const shape_type& shape) {
675 if (shape[0] * shape[1] != _size) {
676 WARN("Reshaping to a different size will cause data loss");
677 }
678 auto* temp = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
679 CHECK(cudaMemcpy(temp, _data, _size * sizeof(value_type), cudaMemcpyDeviceToHost));
680 CHECK(cudaFree(_data));
681
682 value_type* temp_grad = nullptr;
683 if (_requires_grad) {
684 temp_grad = static_cast<value_type*>(malloc(_size * sizeof(value_type)));
685 CHECK(cudaMemcpy(temp_grad, _grad, _size * sizeof(value_type), cudaMemcpyDeviceToHost));
686 CHECK(cudaFree(_grad));
687 }
688
689 const size_type size = _size;
690 _size = shape[0] * shape[1];
691 _shape = shape;
692
693 CHECK(cudaMalloc(&_data, _size * sizeof(value_type)));
694 CHECK(cudaMemset(_data, 0, _size * sizeof(value_type)));
695 CHECK(cudaMemcpy(_data, temp, size * sizeof(value_type), cudaMemcpyHostToDevice));
696 free(temp);
697
698 if (_requires_grad) {
699 CHECK(cudaMalloc(&_grad, _size * sizeof(value_type)));
700 CHECK(cudaMemset(_grad, 0, _size * sizeof(value_type)));
701 CHECK(cudaMemcpy(_grad, temp_grad, size * sizeof(value_type), cudaMemcpyHostToDevice));
702 free(temp_grad);
703 }

706 void Tensor::transpose() {
707 const dim3 block(TILE_SIZE, TILE_SIZE);
708 const dim3 grid((_shape[0] + block.x - 1) / block.x, (_shape[1] + block.y - 1) / block.y);
709 value_type* temp;
710 CHECK(cudaMalloc(&temp, _size * sizeof(value_type)));
711 krnl::Transpose(grid, block, _data, temp, _shape[0], _shape[1]);
712 CHECK(cudaDeviceSynchronize());
713 CHECK(cudaFree(_data));
714 _data = temp;
715 if (_requires_grad) {
716 value_type* tempGrad;
717 CHECK(cudaMalloc(&tempGrad, _size * sizeof(value_type)));
718 krnl::Transpose(grid, block, _grad, tempGrad, _shape[0], _shape[1]);
719 CHECK(cudaDeviceSynchronize());
720 CHECK(cudaFree(_grad));
721 _grad = tempGrad;
722 }
723 std::swap(_shape[0], _shape[1]);
724 }
