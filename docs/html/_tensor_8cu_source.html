378 void Tensor::transpose() {
379 const dim3 block(TILE_SIZE, TILE_SIZE);
380 const dim3 grid((_shape[2] + block.x - 1) / block.x, (_shape[3] + block.y - 1) / block.y);
381 value_type* temp;
382 std::vector<size_t> offset;
383 CHECK(cudaMalloc(&temp, _size * sizeof(value_type)));
384 for (auto i = 0; i < _shape[0]; i += 1) {
385 for (auto j = 0; j < _shape[1]; j += 1) {
386 offset.push_back(i * _shape.getStride(0) + j * _shape.getStride(1));
387 }
388 }
389 krnl::Transpose(grid, block, _data, temp, _shape[2], _shape[3], offset);
390 cuStrm::StreamManager<value_type>::Instance().freeAsync(_data);
391 _data = temp;
392 if (_requires_grad) {
393 value_type* tempGrad;
394 CHECK(cudaMalloc(&tempGrad, _size * sizeof(value_type)));
395 krnl::Transpose(grid, block, _grad, tempGrad, _shape[2], _shape[3], offset);
396 cuStrm::StreamManager<value_type>::Instance().freeAsync(_grad);
397 _grad = tempGrad;
398 }
399 std::swap(_shape[2], _shape[3]);
400 _shape.updateStride();
401 }
