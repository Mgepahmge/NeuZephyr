385 void Tensor::transpose() {
386 const dim3 block(TILE_SIZE, TILE_SIZE);
387 const dim3 grid((_shape[2] + block.x - 1) / block.x, (_shape[3] + block.y - 1) / block.y);
388 value_type* temp;
389 std::vector<size_t> offset;
390 cuStrm::StreamManager<float>::Instance().malloc(&temp, _size * sizeof(value_type));
391 for (auto i = 0; i < _shape[0]; i += 1) {
392 for (auto j = 0; j < _shape[1]; j += 1) {
393 offset.push_back(i * _shape.getStride(0) + j * _shape.getStride(1));
394 }
395 }
396 krnl::Transpose(grid, block, _data, temp, _shape[2], _shape[3], offset);
397 cuStrm::StreamManager<value_type>::Instance().freeAsync(_data);
398 _data = temp;
399 if (_requires_grad) {
400 value_type* tempGrad;
401 cuStrm::StreamManager<float>::Instance().malloc(&tempGrad, _size * sizeof(value_type));
402 krnl::Transpose(grid, block, _grad, tempGrad, _shape[2], _shape[3], offset);
403 cuStrm::StreamManager<value_type>::Instance().freeAsync(_grad);
404 _grad = tempGrad;
405 }
406 std::swap(_shape[2], _shape[3]);
407 _shape.updateStride();
408 }

506 bool Tensor::operator==(const Tensor& other) const {
507 if (_requires_grad != other._requires_grad) {
508 return false;
509 }
510 if (_shape != other._shape) {
511 return false;
512 }
513 constexpr value_type abs_epsilon = 1e-6f;
514 constexpr value_type rel_epsilon = 1e-5f;
515 this->sync();
516 other.sync();
517 auto* temp = new value_type[_size];
518 auto* temp_other = new value_type[_size];
519 cudaMemcpy(temp, _data, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
520 cudaMemcpy(temp_other, other._data, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
521 for (auto i = 0; i < _size; i++) {
522 const auto a = temp[i];
523 const auto b = temp_other[i];
524 const auto diff = std::abs(a - b);
525 if (diff > std::max(rel_epsilon * std::max(std::abs(a), std::abs(b)), abs_epsilon)) {
526 delete[] temp;
527 delete[] temp_other;
528 return false;
529 }
530 }
531 if (_requires_grad) {
532 cudaMemcpy(temp, _grad, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
533 cudaMemcpy(temp_other, other._grad, _size * sizeof(value_type), cudaMemcpyDeviceToHost);
534 for (auto i = 0; i < _size; i++) {
535 const auto a = temp[i];
536 const auto b = temp_other[i];
537 const auto diff = std::abs(a - b);
538 if (diff > std::max(rel_epsilon * std::max(std::abs(a), std::abs(b)), abs_epsilon)) {
539 delete[] temp;
540 delete[] temp_other;
541 return false;
542 }
543 }
544 }
545 delete[] temp;
546 delete[] temp_other;
547 return true;
548 }

664 if (hData[i] == value) {
665 index = i;
666 break;
667 }
668 }
669 auto n = index / (_shape[1] * _shape[2] * _shape[3]);

671 auto h = (index % (_shape[2] * _shape[3])) / _shape[3];
672 auto w = index % _shape[3];
673 return {n, c, h, w};
674 }

676 Tensor::shape_type Tensor::find(value_type value, size_type batch, size_type channel) const {
677 if (batch >= _shape[0] || channel >= _shape[1]) {
678 throw std::invalid_argument("Invalid position");
679 }
680 auto hData = hostData();
681 auto index = 0;
682 auto offset = batch * _shape.getStride(0) + channel * _shape.getStride(1);
683 for (auto i = 0; i < _shape[2] * _shape[3]; ++i) {
684 if (hData[offset + i] == value) {
685 index = i;
686 break;
687 }
688 }
689 auto h = index / _shape[3];
690 auto w = index % _shape[3];
691 return {batch, channel, h, w};
692 }

700 krnl::SummationExp(grid, block, block.x / WARP_SIZE * sizeof(float), dData, _data, _size);

706 result += hData[i];

708 delete[] hData;
709 cuStrm::StreamManager<value_type>::Instance().free(dData);
