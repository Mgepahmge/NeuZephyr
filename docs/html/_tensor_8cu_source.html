727 Tensor Softmax(const Tensor& tensor) {
728 const dim3 block(256);
729 const dim3 grid((tensor._size + block.x - 1) / block.x);
730 float* result_d;
731 float* result_h;
732 float sum = 0;
733 cudaMalloc(&result_d, grid.x * sizeof(Tensor::value_type));
734 result_h = static_cast<float*>(malloc(grid.x * sizeof(Tensor::value_type)));
735 krnl::SummationExp(grid, block, block.x * sizeof(float), result_d, tensor._data, tensor._size);
736 CHECK(cudaMemcpy(result_h, result_d, grid.x * sizeof(Tensor::value_type), cudaMemcpyDeviceToHost));
737 for (int i = 0; i < grid.x; i++) {
738 sum += result_h[i];
739 }
740 free(result_h);
741 cudaFree(result_d);
742 krnl::Softmax(grid, block, tensor._data, tensor._data, sum, tensor._size);
743 CHECK(cudaDeviceSynchronize());
744 return tensor;
745 }

992 WARN("Reshaping to a different size will cause data loss");

1005 const size_type size = _size;

1010 CHECK(cudaMemset(_data, 0, _size * sizeof(value_type)));

1022 void Tensor::transpose() {
1023 const dim3 block(TILE_SIZE, TILE_SIZE);
1024 const dim3 grid((_shape[0] + block.x - 1) / block.x, (_shape[1] + block.y - 1) / block.y);
1025 value_type* temp;
1026 CHECK(cudaMalloc(&temp, _size * sizeof(value_type)));
1027 krnl::Transpose(grid, block, _data, temp, _shape[0], _shape[1]);
1028 CHECK(cudaDeviceSynchronize());
1029 CHECK(cudaFree(_data));
1030 _data = temp;
1031 if (_requires_grad) {
1032 value_type* tempGrad;
1033 CHECK(cudaMalloc(&tempGrad, _size * sizeof(value_type)));
1034 krnl::Transpose(grid, block, _grad, tempGrad, _shape[0], _shape[1]);
1035 CHECK(cudaDeviceSynchronize());
1036 CHECK(cudaFree(_grad));
1037 _grad = tempGrad;
1038 }
1039 std::swap(_shape[0], _shape[1]);
1040 }
