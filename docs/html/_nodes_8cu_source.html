103 void MatMulNode::forward() {
104 // M = A.shape()[0] N = B.shape()[1], K = A.shape()[1]
105 TensorCoreGEMM(inputs[0]->output->data(), inputs[1]->output->data(), output->data(),
106 inputs[0]->output->shape()[0], inputs[1]->output->shape()[1], inputs[0]->output->shape()[1]);
107 }
108
109 void MatMulNode::backward() {
110 // dA = dC * B^T
111 if (inputs[0]->output->requiresGrad()) {
112 Tensor B_T(*inputs[1]->output); // B
113 B_T.transpose(); // B^T
114 TensorCoreGEMM(output->grad(),
115 B_T.data(),
116 inputs[0]->output->grad(),
117 output->shape()[0],
118 B_T.shape()[1],
119 output->shape()[1]);
120 }
121 // dB = A^T * dC
122 if (inputs[1]->output->requiresGrad()) {
123 Tensor A_T(*inputs[0]->output); // A
124 A_T.transpose(); // A^T
125 // M = A.shape()[0] N = B.shape()[1], K = A.shape()[1]
126 TensorCoreGEMM(A_T.data(),
127 output->grad(),
128 inputs[1]->output->grad(),
129 A_T.shape()[0],
130 output->shape()[1],
131 A_T.shape()[1]);
132 }

447 SoftmaxNode::SoftmaxNode(Node* input) {
448 sum = 0;
449 inputs.push_back(input);
450 bool requires_grad = input->output->requiresGrad();
451 output = std::make_shared<Tensor>(input->output->shape(), requires_grad);
452 type = "Softmax";

455 void SoftmaxNode::forward() {
456 const dim3 block(256);
457 const dim3 grid((output->size() + block.x - 1) / block.x);
458 float* result;
459 cudaMalloc((float**)&result, grid.x * sizeof(float));
460 auto* result_host = static_cast<float*>(malloc(grid.x * sizeof(float)));
461 SummationExp(grid, block, block.x / WARP_SIZE * sizeof(float), result, inputs[0]->output->data(),
462 output->size());
463 cudaMemcpy(result_host, result, grid.x * sizeof(float), cudaMemcpyDeviceToHost);
464 for (int i = 0; i < grid.x; i++) {
465 sum += result_host[i];
466 }
467 cudaFree(result);
468 free(result_host);
469 dim3 block2(256);
470 dim3 grid2((output->size() + block.x - 1) / block.x);
471 Softmax(grid2, block2, output->data(), inputs[0]->output->data(), sum, output->size());
472 }
473
474 void SoftmaxNode::backward() {
475 const Tensor jacobian(std::vector<int>({output->shape()[0], output->shape()[0]}), false);
476 const dim3 block(16, 16);
477 const dim3 grid((output->shape()[0] + block.x - 1) / block.x, (output->shape()[0] + block.y - 1) / block.y);
478 SoftmaxJacobian(grid, block, jacobian.data(), output->data(), output->size());
479 TensorCoreGEMM(jacobian.data(), output->grad(), inputs[0]->output->grad(),
480 jacobian.shape()[0], output->shape()[1], jacobian.shape()[1]);
481 }
