100 if (inputs[0]->output->requiresGrad()) {
101 if (inputs[0]->output->shape() == output->shape()) {
102 cuStrm::StreamManager<Tensor::value_type>::Instance().memcpy(

106 else {
107 const dim3 block(BLOCKSIZE);
108 const dim3 grid((output->shape()[2] * output->shape()[3] + BLOCKSIZE - 1) / BLOCKSIZE);
109 std::vector<size_t> offset_o;
110 std::vector<size_t> offset_i;
111 for (auto i = 0; i < output->shape()[0]; i++) {
112 for (auto j = 0; j < output->shape()[1]; j++) {
113 offset_i.push_back(i * output->shape().getStride(0) + j * output->shape().getStride(1));
114 offset_o.push_back(
115 i * (inputs[0]->output->shape()[0] > 1 ? inputs[0]->output->shape().getStride(0) : 0) +
116 j * (inputs[0]->output->shape()[1] > 1 ? inputs[0]->output->shape().getStride(1) : 0));
117 }
118 }

120 output->shape()[2] * output->shape()[3], offset_o, offset_i);
121 }
122 }
123 if (inputs[1]->output->requiresGrad()) {
124 if (inputs[1]->output->shape() == output->shape()) {
125 cuStrm::StreamManager<Tensor::value_type>::Instance().memcpy(
126 inputs[1]->output->grad(), output->grad(), output->size() * sizeof(Tensor::value_type),
127 cudaMemcpyDeviceToDevice);
128 }
129 else {

169 void MatMulNode::backward() {
170 // dA = dC * B^T
171 if (inputs[0]->output->requiresGrad()) {
172 auto B_T = transpose(*inputs[1]->output);
173 iGEMMBackward(output->grad(), B_T.data(), inputs[0]->output->grad(), output->shape(), B_T.shape(),
174 inputs[0]->output->shape());
175 }
176 // dB = A^T * dC
177 if (inputs[1]->output->requiresGrad()) {
178 auto A_T = transpose(*inputs[0]->output);
179 iGEMMBackward(A_T.data(), output->grad(), inputs[1]->output->grad(), A_T.shape(), output->shape(),
180 inputs[1]->output->shape());
181 }

573 ExpandNode::ExpandNode(Node* input, const Tensor::size_type newBatch) : newBatch(newBatch) {
574 if (input->output->shape()[0] != 1) {
575 throw std::invalid_argument("ExpandNode: input must have batch size 1");
576 }
577 inputs.push_back(input);
578 output = std::make_shared<Tensor>(Tensor::shape_type{
579 newBatch, input->output->shape()[1], input->output->shape()[2],
580 input->output->shape()[3]
581 }, input->output->requiresGrad());
582 type = "Expand";
583 }
584
585 void ExpandNode::forward() {
586 const auto size = inputs[0]->output->shape()[1] * inputs[0]->output->shape()[2] *
587 inputs[0]->output->shape()[3];
588 const auto total = size * newBatch;
589 const dim3 block(BLOCKSIZE);
590 const dim3 grid((total + block.x - 1) / block.x);
591 Expand(grid, block, output->data(), inputs[0]->output->data(), size, total);
592 }
593
594 void ExpandNode::backward() {
595 if (inputs[0]->output->requiresGrad()) {
596 const auto size = inputs[0]->output->shape()[1] * inputs[0]->output->shape()[2] *
597 inputs[0]->output->shape()[3];
598 const auto total = size * newBatch;
599 const dim3 block(BLOCKSIZE);
600 const dim3 grid((total + block.x - 1) / block.x);
601 Compress(grid, block, inputs[0]->output->grad(), output->grad(), size, total);
602 }
