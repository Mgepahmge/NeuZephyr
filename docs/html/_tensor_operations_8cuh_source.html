353 template <typename T>
354 T Softmax(T input) {
355 T result(input.shape(), input.requiresGrad());
356 auto size = input.shape()[2] * input.shape()[3];
357 std::vector<size_t> offset;
358 std::vector<float> sum;
359 for (auto i = 0; i < input.shape()[0]; i++) {
360 for (auto j = 0; j < input.shape()[1]; j++) {
361 offset.push_back(i * input.shape().getStride(0) + j * input.shape().getStride(1));
362 sum.push_back(input.expSum(i, j));
363 }
364 }
365 iSoftmax(result.data(), input.data(), sum, size, offset);
