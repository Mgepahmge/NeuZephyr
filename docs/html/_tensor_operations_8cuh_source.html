352 template <typename T>
353 T Softmax(T input) {
354 T result(input.shape(), input.requiresGrad());
355 auto size = input.shape()[2] * input.shape()[3];
356 std::vector<size_t> offset;
357 std::vector<float> sum;
358 for (auto i = 0; i < input.shape()[0]; i++) {
359 for (auto j = 0; j < input.shape()[1]; j++) {
360 offset.push_back(i * input.shape().getStride(0) + j * input.shape().getStride(1));
361 sum.push_back(input.expSum(i, j));
362 }
363 }
364 iSoftmax(result.data(), input.data(), sum, size, offset);
365 return result;
