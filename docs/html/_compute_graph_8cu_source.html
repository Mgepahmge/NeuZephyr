258 void ComputeGraph::topologicalSort() {
259 sortedNodes.clear();
260 inDegree.clear();
261 adjList.clear();
262
263 for (Node* node : nodes) {
264 if (inDegree.find(node) == inDegree.end()) {
265 inDegree[node] = 0;
266 }
267
268 for (Node* input : node->inputs) {
269 adjList[input].push_back(node);
270 inDegree[node]++;
271 }
272 }
273
274 std::queue<Node*> q;
275 for (Node* node : nodes) {
276 if (inDegree[node] == 0) {
277 q.push(node);
278 }
279 }
280
281 while (!q.empty()) {
282 Node* node = q.front();
283 q.pop();
284 sortedNodes.push_back(node);
285 for (Node* next : adjList[node]) {
286 inDegree[next]--;
287 if (inDegree[next] == 0) {
288 q.push(next);
289 }
290 }
291 }
292
293 if (sortedNodes.size() != nodes.size()) {
294 throw std::runtime_error("Graph has cycle");
295 }
296 }

548 void ComputeGraph::save(const std::string& path) {
549 if (path.empty()) {
550 throw std::runtime_error("Path cannot be empty");
551 }
552 if (sortedNodes.empty()) {
553 throw std::runtime_error("Graph not sorted");
554 }
555
556 std::ofstream out(path);
557 if (!out.is_open()) {
558 throw std::runtime_error("Failed to open file for writing");
559 }
560
561 out << "[\n"; // Start the JSON array
562
563 for (size_t i = 0; i < sortedNodes.size(); ++i) {
564 Node* node = sortedNodes[i];
565
566 out << " {\n"; // Start a node object
567 out << R"( "type": ")" << node->type << "\",\n";
568 out << R"( "name": ")" << nodeRosterReverse[node] << "\",\n";
569
570 // Pre nodes (inputs)
571 out << " \"pre\": [";
572 for (size_t j = 0; j < node->inputs.size(); ++j) {
573 auto input = node->inputs[j];
574 auto index = std::distance(sortedNodes.begin(),
575 std::find(sortedNodes.begin(), sortedNodes.end(), input));
576 out << index;
577 if (j < node->inputs.size() - 1) {
578 out << ", ";
579 }
580 }
581 out << "],\n";
582 out << " \"post\": [";
583 for (size_t j = 0; j < adjList[node].size(); ++j) {
584 auto next = adjList[node][j];
585 auto index = std::distance(sortedNodes.begin(),
586 std::find(sortedNodes.begin(), sortedNodes.end(), next));
587 out << index;
588 if (j < adjList[node].size() - 1) {
589 out << ", ";
590 }
591 }
592 out << "],\n";
593 out << " \"shape\": ["
594 << node->output->shape()[0] << ", "
595 << node->output->shape()[1] << "],\n";
596 out << " \"data\": [";
597 auto* data = new float[node->output->size()];
598 cuStrm::StreamManager<Tensor::value_type>::Instance().memcpy(data, node->output->data(), node->output->size() * sizeof(float), cudaMemcpyDeviceToHost);
599 cuStrm::StreamManager<Tensor::value_type>::Instance().syncData(data);
600 for (size_t j = 0; j < node->output->size(); ++j) {
601 out << data[j];
602 if (j < node->output->size() - 1) {
603 out << ", ";
604 }
605 }
606 out << "],\n";
607 delete[] data;
608 out << " \"requires_grad\": " << (node->output->requiresGrad() ? "true" : "false") << ",\n";
609 if (node->output->requiresGrad()) {
610 out << " \"grad\": [";
611 auto* grad_data = new float[node->output->size()];
612 cuStrm::StreamManager<Tensor::value_type>::Instance().memcpy(grad_data, node->output->grad(), node->output->size() * sizeof(float),
613 cudaMemcpyDeviceToHost);
614 cuStrm::StreamManager<Tensor::value_type>::Instance().syncData(grad_data);
615 for (size_t j = 0; j < node->output->size(); ++j) {
616 out << grad_data[j];
617 if (j < node->output->size() - 1) {
618 out << ", ";
619 }
620 }
621 out << "]\n";
622 delete[] grad_data;
623 }
624 else {
625 out << " \"grad\": []\n";
626 }
627 out << " }";
628 if (i < sortedNodes.size() - 1) {
629 out << ",";
630 }
631 out << "\n";
632 }
633 out << "]\n";
634 out.close();
635 }

637 void ComputeGraph::load(const std::string& path) {
638 if (path.empty()) {
639 throw std::runtime_error("Path cannot be empty");
640 }
641 if (!nodes.empty()) {
642 throw std::runtime_error("Graph already loaded");
643 }
644 std::ifstream in(path);
645 if (!in.is_open()) {
646 throw std::runtime_error("Failed to open file for reading");
647 }
648 int i = 0;
649 std::string line;
650 std::string type;
651 std::string name;
652 std::vector<int> pre;
653 std::vector<int> post;
654 Tensor::shape_type shape;
655 float* data = nullptr;
656 bool requires_grad = false;
657 float* grad = nullptr;
658 bool reading_node = false;
659 while (std::getline(in, line)) {
660 if (line.empty()) {
661 continue;
662 }
663 if (line == "[" && !reading_node) {
664 continue;
665 }
666 if (line == "]" && !reading_node) {
667 break;
668 }
669 if (line.find('{') != std::string::npos && !reading_node) {
670 reading_node = true;
671 continue;
672 }
673 if (line.find("\"type\": ") != std::string::npos && reading_node) {
674 std::string pattern = "\"type\": ";
675 size_t startPos = line.find(pattern);
676 startPos += pattern.length();
677 const size_t firstQuote = line.find('\"', startPos);
678 const size_t secondQuote = line.find('\"', firstQuote + 1);
679 type = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
680 continue;
681 }
682 if (line.find("\"name\": ") != std::string::npos && reading_node) {
683 std::string pattern = "\"name\": ";
684 size_t startPos = line.find(pattern);
685 startPos += pattern.length();
686 const size_t firstQuote = line.find('\"', startPos);
687 const size_t secondQuote = line.find('\"', firstQuote + 1);
688 name = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
689 continue;
690 }
691 if (line.find("\"pre\": ") != std::string::npos && reading_node) {
692 pre.clear();
693 std::string pattern = "\"pre\": ";
694 size_t startPos = line.find(pattern);
695 startPos += pattern.length();
696 const size_t firstQuote = line.find('[', startPos);
697 const size_t secondQuote = line.find(']', firstQuote + 1);
698 std::string pre_str = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
699 std::stringstream ss(pre_str);
700 int val;
701 while (ss >> val) {
702 pre.push_back(val);
703 if (ss.peek() == ',') {
704 ss.ignore();
705 }
706 }
707 continue;
708 }
709 if (line.find("\"post\": ") != std::string::npos && reading_node) {
710 post.clear();
711 std::string pattern = "\"post\": ";
712 size_t startPos = line.find(pattern);
713 startPos += pattern.length();
714 const size_t firstQuote = line.find('[', startPos);
715 const size_t secondQuote = line.find(']', firstQuote + 1);
716 std::string post_str = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
717 std::stringstream ss(post_str);
718 int val;
719 while (ss >> val) {
720 post.push_back(val);
721 if (ss.peek() == ',') {
722 ss.ignore();
723 }
724 }
725 continue;
726 }
727 if (line.find("\"shape\": ") != std::string::npos && reading_node) {
728 std::string pattern = "\"shape\": ";
729 size_t startPos = line.find(pattern);
730 startPos += pattern.length();
731 const size_t firstQuote = line.find('[', startPos);
732 const size_t secondQuote = line.find(']', firstQuote + 1);
733 std::string shape_str = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
734 std::stringstream ss(shape_str);
735 int val;
736 auto index = 0;
737 while (ss >> val) {
738 shape[index] = val;
739 index++;
740 if (ss.peek() == ',') {
741 ss.ignore();
742 }
743 }
744 continue;
745 }
746 if (line.find("\"data\": ") != std::string::npos && reading_node) {
747 data = static_cast<float*>(malloc(sizeof(float) * shape[0] * shape[1]));
748 std::string pattern = "\"data\": ";
749 size_t startPos = line.find(pattern);
750 startPos += pattern.length();
751 const size_t firstQuote = line.find('[', startPos);
752 const size_t secondQuote = line.find(']', firstQuote + 1);
753 std::string data_str = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
754 std::stringstream ss(data_str);
755 float val;
756 i = 0;
757 while (ss >> val) {
758 data[i++] = val;
759 if (ss.peek() == ',') {
760 ss.ignore();
761 }
762 }
763 continue;
764 }
765 if (line.find("\"requires_grad\": ") != std::string::npos && reading_node) {
766 std::string pattern = "\"requires_grad\": ";
767 size_t startPos = line.find(pattern);
768 startPos += pattern.length();
769 if (line.substr(startPos, 4) == "true") {
770 requires_grad = true;
771 }
772 else {
773 requires_grad = false;
774 }
775 continue;
776 }
777 if (line.find("\"grad\": ") != std::string::npos && reading_node && requires_grad) {
778 grad = static_cast<float*>(malloc(sizeof(float) * shape[0] * shape[1]));
779 std::string pattern = "\"grad\": ";
780 size_t startPos = line.find(pattern);
781 startPos += pattern.length();
782 const size_t firstQuote = line.find('[', startPos);
783 const size_t secondQuote = line.find(']', firstQuote + 1);
784 std::string grad_str = line.substr(firstQuote + 1, secondQuote - firstQuote - 1);
785 std::stringstream ss(grad_str);

787 i = 0;
788 while (ss >> val) {
789 grad[i++] = val;
790 if (ss.peek() == ',') {
791 ss.ignore();
792 }
793 }
794 continue;
795 }
796 if (line.find('}') != std::string::npos && reading_node) {
797 reading_node = false;
798 CreateNode(this, type, name, pre, shape, data, requires_grad, grad);
799 free(data);
800 if (requires_grad) {
801 free(grad);
802 }
803 continue;
804 }
805 }
