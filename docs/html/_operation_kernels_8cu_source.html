64 __global__ void GeneralMatrixMulKernel(float* C, const float* A, const float* B,
65 const unsigned long long M,
66 const unsigned long long N,
67 const unsigned long long K,
68 const size_t offset_c,
69 const size_t offset_a,
70 const size_t offset_b) {
71 __shared__ float As[TILE_SIZE][TILE_SIZE];
72 __shared__ float Bs[TILE_SIZE][TILE_SIZE];
73
74 const unsigned long long row = blockIdx.y * blockDim.y + threadIdx.y;
75 const unsigned long long col = blockIdx.x * blockDim.x + threadIdx.x;
76
77 float sum = 0.0f;
78
79 for (int t = 0; t < K; t += TILE_SIZE) {
80 if (row < M && t + threadIdx.x < K)
81 As[threadIdx.y][threadIdx.x] = A[row * K + t + threadIdx.x + offset_a];
82 else
83 As[threadIdx.y][threadIdx.x] = 0.0f;
84
85 if (col < N && t + threadIdx.y < K)
86 Bs[threadIdx.y][threadIdx.x] = B[(t + threadIdx.y) * N + col + offset_b];
87 else
88 Bs[threadIdx.y][threadIdx.x] = 0.0f;
89
90 __syncthreads();
91
92 for (int i = 0; i < TILE_SIZE; i++)
93 sum += As[threadIdx.y][i] * Bs[i][threadIdx.x];
94
95 __syncthreads();
96 }
97
98 if (row < M && col < N)
99 C[row * N + col + offset_c] = sum;
100 }

476 __global__ void SummationExpKernel(float* out, const float* g_data,
477 const unsigned long long n, const size_t offset) {
478 extern __shared__ float sdata[];
479 const unsigned long long idx = blockIdx.x * blockDim.x + threadIdx.x;
480 const unsigned long long tid = threadIdx.x;
481 const unsigned long long warpIdx = tid / WARP_SIZE;
482 const unsigned long long laneIdx = tid % WARP_SIZE;
483 float localSum = 0.0f;
484 if (idx < n) {
485 localSum = __expf(g_data[idx + offset]);
486 }
487 else {
488 localSum = 0.0f;
489 }
490 __syncthreads();
491 // Warp Reduce
492 localSum = warpReduce(localSum);
493 if (laneIdx == 0) {
494 sdata[warpIdx] = localSum;
495 }
496 __syncthreads();
497 localSum = (tid < blockDim.x / WARP_SIZE) ? sdata[laneIdx] : 0.0f;
498 // Block Reduce
499 if (warpIdx == 0) {
500 localSum = warpReduce(localSum);
501 }
502 __syncthreads();
503
504 if (tid == 0) {
505 out[blockIdx.x] = localSum;
506 }
507 }

549 __global__ void SoftmaxJacobianKernel(float* out, const float* in,
550 const unsigned long long n) {
551 unsigned long long idx = blockIdx.x * blockDim.x + threadIdx.x;
552 unsigned long long idy = blockIdx.y * blockDim.y + threadIdx.y;
553 if (idx >= n || idy >= n) {
554 return;
555 }
556 unsigned long long id = idx * n + idy;
557 if (idy == idx) {
558 out[id] = in[idx] * (1 - in[idx]);
559 }
560 else {
561 out[id] = -in[idx] * in[idy];
562 }
563 }

744 __global__ void AdamKernel(float* data, float* m, float* v, const float* grad, const float lr, const float beta1,
745 const float beta2, const float eps, const int t, const unsigned long long n) {
746 const unsigned long long idx = blockIdx.x * blockDim.x + threadIdx.x;
747 if (idx >= n) {
748 return;
749 }
750 const float m_temp = m[idx] * beta1 + grad[idx] * (1 - beta1);
751 const float v_temp = v[idx] * beta2 + grad[idx] * grad[idx] * (1 - beta2);
752 const float m_modified = m_temp / (1 - __powf(beta1, (float)t));
753 const float v_modified = v_temp / (1 - __powf(beta2, (float)t));
754 data[idx] -= lr * m_modified / (sqrtf(v_modified) + eps);
755 m[idx] = m_temp;
756 v[idx] = v_temp;
757 }

793 const unsigned long long n) {

795 if (idx >= n) {

798 const float delta_acc_grad_temp = acc_grad[idx] * rho + grad[idx] * grad[idx] * (1 - rho);

800 data[idx] += delta_theta;

802 acc_delta[idx] = delta_acc_temp;

812 __global__ void GeneralMatrixMulTensorKernel(float* C, const half* A, const half* B, const unsigned long long m,
813 const unsigned long long n, const unsigned long long k) {
814 const unsigned long long idx = blockIdx.x * blockDim.x + threadIdx.x;
815 const unsigned long long warpIdx = idx / warpSize;
816 const unsigned long long blockM = m / MMA;
817 const unsigned long long blockN = n / MMA;
818 const unsigned long long rowA = warpIdx / blockN;
819 const unsigned long long colB = warpIdx % blockN;
820 nvcuda::wmma::fragment<nvcuda::wmma::matrix_a, MMA, MMA, MMA, half, nvcuda::wmma::row_major> A_frag;
821 nvcuda::wmma::fragment<nvcuda::wmma::matrix_b, MMA, MMA, MMA, half, nvcuda::wmma::row_major> B_frag;
822 nvcuda::wmma::fragment<nvcuda::wmma::accumulator, MMA, MMA, MMA, float> C_frag;
823 fill_fragment(C_frag, 0.0f);
824 if (rowA < blockM && colB < blockN) {
825 for (int i = 0; i < k / MMA; i++) {
826 load_matrix_sync(A_frag, A + rowA * k * MMA + i * MMA, k);
827 load_matrix_sync(B_frag, B + colB * MMA + i * n * MMA, n);
828 mma_sync(C_frag, A_frag, B_frag, C_frag);
829 }
830 store_matrix_sync(C + rowA * n * MMA + colB * MMA, C_frag, n, nvcuda::wmma::mem_row_major);

859 void TensorCoreGEMM(float* A, float* B, float* C, const unsigned long long M,
860 const unsigned long long N, const unsigned long long K) {
861 const unsigned long long m = CEIL(M);
862 const unsigned long long k = CEIL(K);
863 const unsigned long long n = CEIL(N);
864 half* padded_A;
865 half* padded_B;

868 StreamManager<float>::Instance().malloc(&padded_B, k * n * sizeof(half));
869 StreamManager<float>::Instance().malloc(&padded_C, m * n * sizeof(float));
870 StreamManager<float>::Instance().memset(padded_C, 0, m * n * sizeof(float));
871 StreamManager<float>::Instance().submit(Padding, dim3((m * k + 256 - 1) / 256), dim3(256), 0, padded_A, A, M, K,
872 m, k);
873 StreamManager<float>::Instance().submit(Padding, dim3((k * n + 256 - 1) / 256), dim3(256), 0, padded_B, B, K, N,
874 k, n);
875 const unsigned long long tiles = (m * n) >> 8;
876 dim3 block(256);
877 const unsigned int warpPerBlock = block.x / 32;
878 dim3 grid((tiles + warpPerBlock - 1) / warpPerBlock);
879 StreamManager<float>::Instance().submit(GeneralMatrixMulTensorKernel, grid, block, 0, padded_C, padded_A,
880 padded_B, m, n, k);
881 StreamManager<float>::Instance().submit(Cutting, dim3((m * n + 256 - 1) / 256), dim3(256), 0, C, padded_C, M, N,
882 m, n);
883 StreamManager<float>::Instance().freeAsync(padded_A);
884 StreamManager<float>::Instance().freeAsync(padded_B);
885 StreamManager<float>::Instance().freeAsync(padded_C);
886 }
