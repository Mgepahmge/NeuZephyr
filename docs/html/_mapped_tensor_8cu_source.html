374 void MappedTensor::randomize(size_type seed, const bool isGrad) const {
375 if (isGrad && !_requires_grad) {
376 throw std::invalid_argument(
377 "Gradient randomization is not allowed for tensors that do not require gradients.");
378 }
379 if (!seed) {
380 seed = std::chrono::system_clock::now().time_since_epoch().count();
381 }
382 curandGenerator_t generator;
383 curandStatus_t status = curandCreateGenerator(&generator, CURAND_RNG_PSEUDO_DEFAULT);
384 if (status != CURAND_STATUS_SUCCESS) {
385 throw std::runtime_error("Failed to create CURAND generator.");
386 }
387 status = curandSetPseudoRandomGeneratorSeed(generator, seed);
388 if (status != CURAND_STATUS_SUCCESS) {
389 throw std::runtime_error("Failed to set CURAND seed.");
390 }
391 status = curandGenerateUniform(generator, isGrad ? _grad : _data, _size);
392 if (status != CURAND_STATUS_SUCCESS) {
393 throw std::runtime_error("Failed to generate random numbers.");
394 }
395 }

421 void MappedTensor::transpose() {
422 const dim3 block(TILE_SIZE, TILE_SIZE);
423 const dim3 grid((_shape[2] + block.x - 1) / block.x, (_shape[3] + block.y - 1) / block.y);
424 value_type* temp;
425 std::vector<size_t> offset;
426 CHECK(cudaMallocHost(&temp, _size * sizeof(value_type)));
427 for (auto i = 0; i < _shape[0]; i += 1) {
428 for (auto j = 0; j < _shape[1]; j += 1) {
429 offset.push_back(i * _shape.getStride(0) + j * _shape.getStride(1));
430 }
431 }
432 krnl::Transpose(grid, block, _data, temp, _shape[2], _shape[3], offset);
433 cuStrm::StreamManager<value_type>::Instance().freeHost(_data);
434 _data = temp;
435 if (_requires_grad) {
436 value_type* tempGrad;
437 CHECK(cudaMallocHost(&tempGrad, _size * sizeof(value_type)));
438 krnl::Transpose(grid, block, _grad, tempGrad, _shape[2], _shape[3], offset);
439 cuStrm::StreamManager<value_type>::Instance().freeHost(_grad);
440 _grad = tempGrad;
441 }
442 std::swap(_shape[2], _shape[3]);
443 _shape.updateStride();
444 }

475 bool MappedTensor::operator==(const MappedTensor& other) const {
476 if (_requires_grad != other._requires_grad) {
477 return false;
478 }
479 if (_shape != other._shape) {
480 return false;
481 }
482 constexpr value_type abs_epsilon = 1e-6f;
483 constexpr value_type rel_epsilon = 1e-5f;
484 this->sync();
485 other.sync();
486 for (auto i = 0; i < _size; i++) {
487 const auto a = _data[i];
488 const auto b = other._data[i];
489 const auto diff = std::abs(a - b);
490 if (diff > std::max(rel_epsilon * std::max(std::abs(a), std::abs(b)), abs_epsilon)) {
491 return false;
492 }
493 }
494 if (_requires_grad) {
495 for (auto i = 0; i < _size; i++) {
496 const auto a = _grad[i];
497 const auto b = other._grad[i];
498 const auto diff = std::abs(a - b);
499 if (diff > std::max(rel_epsilon * std::max(std::abs(a), std::abs(b)), abs_epsilon)) {
500 return false;
501 }
502 }
503 }
504 return true;
505 }
