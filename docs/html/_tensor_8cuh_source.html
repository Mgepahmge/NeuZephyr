496 void dataInject(Iterator begin, Iterator end, const bool grad = false) const {
497 if (std::distance(begin, end) < _size) {
498 throw std::runtime_error("The length of the input array is less than the size of the tensor.");
499 }
500 auto* host_data = new value_type[_size];
